import matplotlib.pyplot as plt
import numpy as np
import seaborn as sns
sns.set(style="ticks", font_scale=1.5)

def budyko_original(DI):
    return (DI * np.tanh(1/DI) * (1-np.exp(-DI)) )**0.5

def fu(DI, omega):
    return 1 + DI - (1 + DI**omega)**(1/omega)
